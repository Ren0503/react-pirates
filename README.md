## Install

```bash
npm install --save react-pirates
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-pirates'
import 'react-pirates/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
